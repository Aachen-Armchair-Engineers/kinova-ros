120 KinovaPose::KinovaPose(const geometry_msgs::Pose &pose)
121 {
122  double tx, ty, tz;
123  tf::Quaternion q;
124  tf::quaternionMsgToTF(pose.orientation, q);
125 
126  tf::Matrix3x3 bt_q(q);
127 
128  // tf::Matrix3x3 EulerYPR = Rz(tz)*Ry(ty)*Rx(tx)
129  bt_q.getEulerYPR(tz, ty, tx);
130 
131  X = static_cast<float>(pose.position.x);
132  Y = static_cast<float>(pose.position.y);
133  Z = static_cast<float>(pose.position.z);
134 
135  ThetaX = normalizeInRads(tx);
136  ThetaY = normalizeInRads(ty);
137  ThetaZ = normalizeInRads(tz);
